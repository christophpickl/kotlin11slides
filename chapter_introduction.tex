\frame{\frametitle{JVM News}
  \begin{itemize}[<+->]
    \item \textbf{Java 8} bytecode via compiler option: \texttt{-jvm-target 1.8} % has no effect yet
    \item \textbf{JDK 8} classes via dependency: \texttt{kotlin-stdlib-jre8} % instead kotlin-stdlib
    \item Preserve \textbf{parameter names} via \texttt{-java-parameters}
    \item \texttt{const val} are now being \textbf{inlined} by the compiler
    \item Structural changes for \texttt{\textbf{kotlin-reflect.jar}} (Java 9)
    \begin{itemize}
	\item \texttt{kotlin.reflect} renamed to \texttt{kotlin.reflect.full}
	\item Old \textbf{deprecated} and to be removed with Kotlin \texttt{1.2}
	% classes moved from kotlin.reflect to kotlin.reflect.full (old are deprecated and will be deleted with kotlin 1.2)
	% core classes like KClass not affected, as part of std lib
	% why => in order to prepare for java9, the jigsaw/modules feature; 
    \end{itemize}
  \end{itemize}
}
